# RNComponent

# 下包
先yarn install 再 pod install

# 列出可用设备列表
xcrun simctl list devices

# 选择要使用的设备（模拟器）
yarn ios --simulator "iPhone 12"

#  安装打包模块
npm install -g exp

# 开始打包
# 打包  android 
exp build:android

# 打包  ios 
exp build:ios
